void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await Firebase.initializeApp(
    options: DefaultFirebaseOptions.currentPlatform,
  );

  if (kDebugMode) {
    try {
      FirebaseAuth.instance.useAuthEmulator('localhost', 4425);
      FirebaseFirestore.instance.useFirestoreEmulator('localhost', 4436);

      SystemChrome.setPreferredOrientations([DeviceOrientation.portraitUp]);
    } catch (e) {
      // ignore: avoid_print
      print(e);
    }
  }

  runApp(const GodlyTouch());
}

class GodlyTouch extends StatelessWidget {
  const GodlyTouch({super.key});

  @override

    return MaterialApp(
      title: 'Eclipse',
      theme: ThemeData(

            Colors.black, // Set scaffold background to black
        canvasColor: Colors.black, // Set canvas color to black
        visualDensity: VisualDensity.adaptivePlatformDensity,
      ),
      home: const LearnTheUser(),
    );
  }
}
